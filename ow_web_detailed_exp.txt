// Detection and Fallback Logic
const detectWebGLCapabilities = () => {
  const capabilities = {
    webgl: false,
    webgl2: false,
    maxTextureSize: 0,
    extensions: []
  };
  
  // WebGL 1 Support
  const canvas = document.createElement('canvas');
  const gl = canvas.getContext('webgl') || canvas.getContext('experimental-webgl');
  
  if (gl) {
    capabilities.webgl = true;
    capabilities.maxTextureSize = gl.getParameter(gl.MAX_TEXTURE_SIZE);
    
    // Check critical extensions
    const requiredExtensions = [
      'OES_texture_float',
      'WEBGL_draw_buffers',
      'EXT_blend_minmax'
    ];
    
    capabilities.extensions = requiredExtensions.filter(ext => 
      gl.getExtension(ext)
    );
  }
  
  return capabilities;
};

// Adaptive Renderer
class AdaptiveSceneRenderer {
  constructor() {
    this.capabilities = detectWebGLCapabilities();
    this.currentMode = this.determineRenderMode();
  }
  
  determineRenderMode() {
    if (!this.capabilities.webgl) return '2d';
    if (this.capabilities.extensions.length < 2) return 'low';
    if (window.devicePixelRatio > 2) return 'medium';
    return 'high';
  }
  
  async initializeScene() {
    switch (this.currentMode) {
      case 'high':
        return new WebGLScene({ quality: 'high' });
      case 'medium':
        return new WebGLScene({ quality: 'medium' });
      case 'low':
        return new Canvas2DScene();
      case '2d':
        return new LottieFallback();
    }
  }
}
